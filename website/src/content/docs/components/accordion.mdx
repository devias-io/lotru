## Usage

```tsx
import {
  Accordion,
  AccordionContent,
  AccordionItem,
  AccordionTrigger,
} from "@/components/ui/accordion";
```

```tsx
<Accordion>
  <AccordionItem value="item-1">
    <AccordionTrigger>Is it responsive?</AccordionTrigger>
    <AccordionContent>Yes. It adapts seamlessly to different screen sizes.</AccordionContent>
  </AccordionItem>
</Accordion>
```
