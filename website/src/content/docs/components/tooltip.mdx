## Usage

```tsx
import { Tooltip } from "@/components/ui/tooltip";
```

```tsx
<TooltipProvider>
  <Tooltip>
    <TooltipTrigger>Hover</TooltipTrigger>
    <TooltipPositioner>
      <TooltipContent>Add to library</TooltipContent>
      <TooltipArrow />
    </TooltipPositioner>
  </Tooltip>
</TooltipProvider>
```
